4. Bookmark Parsing

public static void main(String[] sa)
	throws StreamReadException,DatabindException,IOException {
	parseFirefoxBookmark("e:/firefox_bookmark.json","e:/firefox_bookmark_uri.txt");
}

private static void parseFirefoxBookmark(String json,String urlFile)
	throws IOException,StreamReadException,DatabindException {
	final ObjectMapper om = new ObjectMapper();
	File file = new File(json);
	List<Bookmark> bookmarkList = om.readValue(file,
		new TypeReference<List<Bookmark>>(){});
	var uriList = bookmarkList.stream()
		.map(p -> p.getUri()).collect(Collectors.toList());
	Files.write(Path.of(urlFile),uriList);
}
