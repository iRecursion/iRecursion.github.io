2. ML Dag

from airflow import DAG
from airflow.operators.python import PythonOperator, BranchPythonOperator
from airflow.operators.bash import BashOperator

from datetime import datetime
from random import randint

def _choosing_best_model(ti):
    accuracies = ti.xcom_pull(task_ids=[
        'training_model_A',
        'training_model_B',
        'training_model_C'
        ])
    if max(accuracies) > 8:
        return 'accurate'
    return 'inaccurate'

def _training_model(model):
    return randint(1, 10)

with DAG("airflow-ml",
    start_date=datetime(2021, 1 ,1),
    schedule_interval='*/30 * * * * *',
    catchup=False) as dag:

    training_model_tasks = [
        PythonOperator(
            task_id=f"training_model_{model_id}",
            python_callable=_training_model,
            op_kwargs={
                "model": model_id
            }
        ) for model_id in ['A', 'B', 'C']
    ]

    choosing_best_model = BranchPythonOperator(
        task_id="choosing_best_model",
        python_callable=_choosing_best_model
    )

    accurate = BashOperator(
        task_id="accurate",
        bash_command="echo 'accurate'"
    )

    inaccurate = BashOperator(
        task_id="inaccurate",
        bash_command=" echo 'inaccurate'"
    )

    training_model_tasks >> choosing_best_model >> [accurate, inaccurate]

#! /usr/bin/env python3
# -*- coding: utf-8 -*-
# creator: elf

from airflow import DAG
from datetime import datetime
from datetime import timedelta
import pendulum

import base64
import logging

from airflow.operators.python import PythonOperator
from airflow.providers.ssh.operators.ssh import SSHOperator
from airflow.operators.empty import EmptyOperator

# Exceptions
from airflow.exceptions import AirflowFailException

local_tz = pendulum.timezone("Asia/Shanghai")

"""
decode base64 xcom output from SSHOperator
"""
def decode_msg(task_name, ti):
    msg = ti.xcom_pull(task_ids=task_name)
    print(msg)
    return base64.b64decode(msg).decode()

"""
fail job if decode number less than threshold
"""
def validate_job(task_name, threshold, ti):
    number = int(ti.xcom_pull(task_ids=task_name))
    logging.info(f"Number generated is: {number}")
    if number < threshold:
        raise AirflowFailException(f"Number {number} is smaller than threshold {threshold}, unlucky.")
    else:
        logging.info(f"NUmber {number} is larger than or equal to threshold {threshold}, job continues.")

default_args = {
    'owner': 'airflow',
    'email': ['ns975@126.com'],
    'email_on_retry': False,
    'email_on_failure': True,
    'retries': 1,
    'retry_delay': timedelta(seconds=10),
    'start_date': datetime(2022,7,10,16,0,0,0,local_tz),
    'do_xcom_push': True,
}

with DAG(
    'dag_ssh',
    default_args=default_args,
    start_date=datetime(2022,7,10,16,0,0,0,tzinfo = local_tz),
    schedule_interval=None,
    catchup = False,
) as dag:
    start = EmptyOperator(task_id='start')
    end = EmptyOperator(task_id='end')
    ssh = SSHOperator(
        task_id='ssh',
        ssh_conn_id='connection_ssh',
        command='echo $RANDOM',
    )
    decode = PythonOperator(
        task_id='decode',
        python_callable=decode_msg,
        op_args=[
            'ssh',
        ],
    )
    validate = PythonOperator(
        task_id='validate',
        python_callable=validate_job,
        op_args=[
            'decode',
            10000,
        ]
    )
    start >> ssh >> decode >> validate >> end

def getTableSchema(query_schema_dict):
    url = host + "/telamon/v1/data/jdbc/preview"
    response = requests.post(url, None, query_schema_dict, headers=headers)
    return response.json()["data"]["schema"]

def getGroupSchema(groupprn):
    url = host + "/telamon/v1/tableGroups?prn=" + groupprn
    response = requests.get(url, headers=headers)
    try:
        return response.json()["data"]["list"][0]["schema"]
    except:
        return ["1"]

def is_group_exist(group_name):
    # group_name 数据组名字
    url = host + "/telamon/v1/tableGroups?prn=" + group_name
    response = requests.get(url, headers=headers)
    resp_json = response.json()

    if len(resp_json["data"]["list"]) != 0:
        return True
    else:
        create_group(group_name)

def create_group(group_prn):
    # url抓包
    url = host + "/telamon/v1/table-groups?well=1"
    param_dict = {"prn": group_prn, "catalogPath": "Data/Untitled/CWPZ"}
    bodys = json.dumps(param_dict)
    response = requests.post(url, headers=headers, data=bodys)
    return response

def import_data_fromdatabase(parm_dict):

def diffSchemaQuery(schemaSou , schemaDst):
    dafault_value = {
        "String": " ' ' ",
        "Double": 0.0,
        "BigInt": 0,

    schemaSoutmp = []
    for t in schemaSou:
        schemaSoutmp.append(t.get('name'))

    schemaDsttmp = []
    for t in schemaDst:
        if t.get('name') in schemaSoutmp:
            schemaDsttmp.append(t.get('name'))
        else:
            extr_name = t.get('name')
            extr_name_type = t.get('type')
            extr_name_defvale = dafault_value.get(extr_name_type)
            schemaDsttmp.append(" {} as {}".format(extr_name_defvale, extr_name))

    query1 = ','.join(schemaDsttmp)
    return ' ' + query1 + ' '
