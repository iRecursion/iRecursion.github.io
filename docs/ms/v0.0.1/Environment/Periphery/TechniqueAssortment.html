2. Mvn System

<dependency>
    <groupId>com.elf</groupId>
    <artifactId>system-jar</artifactId>
    <version>1.0.0</version>
    <scope>system</scope>
    <systemPath>${project.basedir}/library/elf.jar</systemPath>
</dependency>

<resources>
    <resource>
        <directory>lib</directory>
        <targetPath>/BOOT-INF/lib/</targetPath>
        <includes>
            <include>**/*.jar</include>
        </includes>
    </resource>
</resources>
