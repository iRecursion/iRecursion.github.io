2. Remove Prior

docker stop cassandra && docker remove cassandra \
	&& rm -rf cassandra/

3. Run Cassandra

docker run \
	--name cassandra \
	--hostname cassandra \
	-p 9042:9042 \
	--privileged=true \
	--net network-common \
	-v /elf/cassandra:/var/lib/cassandra \
	-itd cassandra:5.0.0

4. Connect Cassandra
