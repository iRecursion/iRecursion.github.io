3. Content

  • Private Repository Authentication
    https://docs.antora.org/antora/latest/playbook/private-repository-auth

content:
  branches: [master, dev, stage]
  tags: [release/*, '!release/*-patch']

  # {web_url}，{refname}，{refhash}，{path}
  # 禁用edit_url: ~
  edit_url: '{web_url}/blob/{refname}/{path}'

  sources:
  - url: https://git-service.com/org/repo-z.git
    start_path: path-to/content-source-root
  - url: https://git-service.com/org/repo-y.git

  - url: D:/Install/Space/GitSpace/AsciiDoc/asciidoc-management/
    branches: master

    start_path: management

  - url: D:/Install/Space/GitSpace/AsciiDoc/asciidoc-overtness/
    # 通过模式指定分支：v*，排除分支：!
    branches: [master, dev, v*, !stage, '!v1.*']

	# 参考antora.yaml中name，start_paths: docs/product-*
    start_path: overtness

    # 提供fallback版本
    version: true

    # antora不使用worktree中的文件，指定分支：[., v2.0]
    # 不同分支下不允许相同组件下有相同文件,否则antora无法运行
    worktrees: false

branches:
- 'master'
- development

start_paths:
- asciidoctor
- party

4. Content Key

  • Content：branches，edit_url，sources，tags

  • Source：branches，edit_url，start_path，
    start_paths，tags，url，version，worktrees

content:
  branches: [v2.0, v2.5, v3.0]
  tags: [release/*, '!release/*-patch']
  edit_url: '{web_url}/blob/{refname}/{path}'
  sources:
  - url: https://git-service.com/org/repo-z.git
    branches: [v1.*, v2.*, !v1.2]
    start_paths: path-to/content-source-root-{item..item}
  - url: https://git-service.com/org/repo-y.git
    branches: []
    tags: [release/*, '!release/*-patch']
    start_path: path-to/content-source-root
    edit_url: '{web_url}/blob/{refname}/{path}'
  - url: https://git-service.com/org/repo-x.git
    branches: v*
    version: true

5. Git

  • credentials.contents，credentials.path，
    fetch_concurrency(fetch和clone并发数，默认0)，

  • ensure_git_suffix(默认true)，

  • plugins.credential_manager，plugins.http

git:
  fetch_concurrency: 5
  ensure_git_suffix: false
  credentials:
    path: ./.git-credentials
    contents: https://$GITHUB_TOKEN:@github.com
  plugins:
    http: ./my-git-http-plugin.js
    credential_manager: ./system-git-credential-manager.js

9. Output

  • Playbook默认使用fs provider，并发布到build/site；

  • Antora内置Provider：fs(filesystem)和archive(ZIP)，

  • 禁用发布：将destinations设置为空数组 []

  • dir

    • ~：user’s home directory相对目录

    • .：playbook相对目录

    • ~+：当前工作目录解析路径

    • /：绝对路径，如/elf/antora/asciidoc

  • clean，destinations[n].clean：--clean，
    默认false，每次生成站点前执行递归清理
    destinations[n].path，destinations[n].provider，

output:
  clean: true
  dir: ./public
  destinations:
  - provider: archive
    path: ./public/blue.zip
  - provider: fs
    path: ./distribution
    clean: true

10. Runtime

  • cache_dir：Win环境：$APPDATA/antora/Caches；
    Linux环境：$XDG_CACHE_HOME/antora；
    若未设置$XDG_CACHE_HOME，则 $HOME/.cache/antora

Key Default Value

cache_dir

fetch

true，每次运行antora，将刷新cache

log.failure_level

非零退出，warn，error，fatal(默认)，none

log.format

label或json(默认)

log.level

all，debug，info，warn(默认)，error，fatal，silent(关闭日志)

log.level_format

当 log.format 为json，接受值label(默认)和number

log.destination.file

stdout 或 stderr

log.destination.append

false

log.destination.buffer_size

0

log.destination.sync

true

 
runtime:
  cache_dir: ./.cache/antora
  fetch: true
  log:
    format: json
    failure_level: warn
    level: error
    level_format: number
    destination:
      file: ./antora.log
      append: true
      buffer_size: 4096
      sync: false
# https://jqlang.github.io/jq
antora --log-level=debug --log-format=json --fetch antora-playbook.yml | jq
antora --log-level=debug --log-format=json antora-playbook.yml | jq

antora --log-level=error antora-playbook.yml | \
	jq 'select(.name == "asciidoctor" and (.msg | contains(" not found:")))'
