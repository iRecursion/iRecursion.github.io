1. Session Mode

version: "2.2"
services:
  jobmanager:
    image: flink:1.19-java17
    ports:
      - "8081:8081"
    command: jobmanager
    environment:
      - |
        FLINK_PROPERTIES=
        jobmanager.rpc.address: jobmanager

  taskmanager:
    image: flink:1.19-java17
    depends_on:
      - jobmanager
    command: taskmanager
    scale: 1
    environment:
      - |
        FLINK_PROPERTIES=
        jobmanager.rpc.address: jobmanager
        taskmanager.numberOfTaskSlots: 2

2. General

Feature Command

Launch Cluster

docker-compose up -d

Stop Cluster

docker-compose stop

Start Cluster

docker-compose start

Kill Cluster

docker-compose down

Access WebUI

http://192.168.0.123:8081

JobManager Logger

docker-compose logs -f jobmanager

TaskManager Logger

docker-compose logs -f taskmanager

TaskManager Recovery

docker-compose up -d taskmanager

Mock Failure

docker-compose kill taskmanager

3. Scale TaskManager

# scale cluster up or down to N TaskManager
docker-compose scale taskmanager=<N>

4. Access JobManager

docker exec -it $(docker ps --filter name=jobmanager --format={{.ID}}) /bin/sh
