1. Entry

1.1. Enable Extension

/* 在每一个数据库执行一次 */
create extension vector;

1.2. Create Vector

/* create vector column with 3 dimension */

create table order_item (
	id bigserial primary key,
	embedding vector(3)
);

1.3. Insert Vector

insert into order_item (embedding) values ('[1,2,3]'),('[4,5,6]');

1.4. Nearest Neighbor

/* get nearest neighbor by L2 distance */
select * from order_item order by embedding <-> '[3,1,2]' limit 5;

  • 支持inner product (<#>) and cosine distance (<⇒)

  • <#> 返回 negative inner product，
    因Postgres在operator上只支持asc order index scan

2. Storing

2.1. Vector Column

/* create table with vector column */

create table order_item(
	id bigserial primary key,
	embedding vector(3)
);

/* add vector column to existing table */
alter table order_item add column embedding vector(3);

2.2. Insert Vector

insert into order_item(embedding) values ('[1,2,3]'),('[4,5,6]');

2.3. Load Vector

/* 或load vector in bulk using copy(example) */
copy order_item(embedding) from stdin with(format binary);

2.4. Upsert Vector

insert into order_item(id, embedding)
	 values (1, '[1,2,3]'),(2, '[4,5,6]') on conflict
	 (id) do update set embedding = excluded.embedding;

2.5. Update Vector

update order_item set embedding = '[1,2,3]' where id = 1;

2.6. Delete Vector

delete from order_item where id = 1;

3. Distance

3.1. Get Distance

select embedding <-> '[3,1,2]' as distance from order_item;

3.2. Inner Product

For

/* inner product multiply by -1， <#> 返回negative inner product) */

select (embedding <#> '[3,1,2]') * -1 as inner_product from order_item;

3.3. Cosine Similarity

/* cosine similarity，use 1 - cosine distance */

select 1 - (embedding <=> '[3,1,2]') as cosine_similarity from order_item;

4. Aggregate

4.1. Average Vector

select avg(embedding) from order_item;

4.2. Average Vector Group

select category_id, avg(embedding) from order_item group by category_id;

5. Indexing

  • 精确的最近邻居搜索：exact nearest neighbor search，
    近似最近邻搜索：approximate nearest neighbor search

  • Postgres Vector默认执行精确的最近邻居搜索，提供完美召回recall，
    可添加索引来使用近似最近邻搜索，用召回换取速度(trade recall for speed)

  • 与典型(typical)索引不同，添加近似(approximate)索引后，
    将看到不同的搜索结果，支持的索引类型：HNSW和IVFFlat；

6. HNSW

6.1. HNSW Entry

  • HNSW索引创建多层图multilayer graph，它具有比IVFFlat更好的查询性能
    (在权衡速度 - 召回方面in term of speed-recall tradeoff)，
    但构建时间较慢slower build time，使用更多内存use more memory；

  • 另因没像IVFFlat这样的训练步骤，故可在表中没有任何数据的情况下创建索引；

  • 可对多达2000个维度的向量进行索引；

/* L2 distance */
create index on order_item using hnsw (embedding vector_l2_ops);

/* inner product */
create index on order_item using hnsw (embedding vector_ip_ops);

/* cosine distance */
create index on order_item using hnsw (embedding vector_cosine_ops);

6.2. Index Option

  • 指定HNSW参数：

    • m：每层的最大连接数，默认16

    • ef_construction：构建图的动态候选列表的大小，默认64，
      dynamic candidate list size for constructing graph

  • ef_construction值越高，以
    index build time / insert speed 为代价cost提供更好的recall。

create index on order_item using hnsw (embedding vector_l2_ops)
	with (m = 16, ef_construction = 64);

6.3. Query Option

  • 指定要搜索的动态候选列表的大小，默认40，
    dynamic candidate list size for search

  • 更高的值以速度为代价提供更好的recall，
    higher value provide better recall at cost of speed

﻿* 在事务内部使用set local为单个查询设置它，
set local inside transaction to set it for single query

/* 指定要搜索的动态候选列表的大小 */
set hnsw.ef_search = 100;

begin;
set local hnsw.ef_search = 100;
select ……
commit;

6.4. Index Build Time

  • 当图适合fit maintenance_work_mem时，索引构建速度明显significantly加快；

set maintenance_work_mem = '8GB';

  • 当图不在适合fit时，会显示一条通知：

Entry Memo

NOTICE

在100000个tuple后，hnsw图不再fit maintenance_work_mem

DETAIL

Building将花费更多take significantly的时间

HINT

增加maintenance_work_mem以加快构建速度

  • 不要将maintenance_work_mem设置得太高，以免耗尽exhaust服务器内存；

  • 与其它索引类型一样，在加载初始数据后创建索引会更快；
    从v0.6.0开始，可通过增加parallel worker(默认2个)来加快索引创建速度；

  • 对large number worker，还需增加max_paralle_workers，默认8；

-- plus leader
set max_parallel_maintenance_workers = 7;

6.5. Indexing Progress

  • HNSW phase：initializing，loading tuple；

select phase,round(100.0 * blocks_done / nullif(blocks_total,0),1)
	as "%" from pg_stat_progress_create_index;

7. IVFFlat

  • IVFFlat索引将向量划分为divide into多个列表，
    然后搜索这些列表中最接近closest查询向量的子集；

  • 与HNSW相比，它具有更快的构建时间和更少的内存使用，但查询性能较低
    (权衡召回速度方面，in terms of speed-recall tradeoff)；

  • 实现良好召回recall的三个关要素：

    • 在表中有一些数据后创建索引；
      create index after table has some data

    • 查询时，指定适当数量的probe，越高对recall越好，
      越低对speed越好，最好从sqrt(listSize)开始；
      https://github.com/pgvector/pgvector?tab=readme-ov-file#query-options

    • 选择适当数量的列表，好的开始位置：

Formula Memo

rowCount / 1000

up to 1M row

sqrt(rowCount)

over 1M row

  • 为要使用的每个距离函数each distance function添加一个索引，
    可对多达2000个维度dimension的向量vector进行索引；

-- L2 distance
create index on order_item using
	ivfflat (embedding vector_l2_ops) with (lists = 100);

-- inner product
create index on order_item using
	ivfflat (embedding vector_ip_ops) with (lists = 100);

-- cosine distance
create index on order_item using
	ivfflat (embedding vector_cosine_ops) with (lists = 100);

7.1. Query Option

  • 指定probe数，默认为1；

set ivfflat.probes = 10;

  • 精确近邻搜索：exact nearest neighbor search

  • 更高的值以速度为代价提供更好的召回recall，且可将其设置为精确近邻搜索的列表数量，
    此时planner不会使用索引，at which point planner won’t use index；

  • 为单个查询在事务内部设置：

begin;
set local ivfflat.probes = 10;
select ……
commit;

7.2. Index Build Time

  • 增加parallel worker number(默认为2)以加快speed up大型表上的索引创建速度；

﻿* 对large worker number，还需增加max_parallel_workers，默认为8；

-- plus leader
set max_parallel_maintenance_workers = 7;

7.3. Indexing Progress

  • IVFFlat phase：

    • initializing

    • performing k-means

    • assigning tuple

    • loading tuple

  • % only populate during loading tuple phase

select phase,
	round(100.0 * tuples_done / nullif(tuples_total,0),1) as "%"
	from pg_stat_progress_create_index;

8. Filtering

8.1. Where Nearest Neighbor

-- 用where子句为最近邻居nearest neighbor查询建立索引：

select * from order_item where category_id = 123
	order by embedding <-> '[3,1,2]' limit 5;

8.2. Where Exact Search

-- 在一或多个where列创建索引以精确搜索exact search：

create index on order_item(category_id);

8.3. Partial Index

-- 或向量列上的部分索引partial index以近似approximate搜索：

create index on order_item using
	hnsw(embedding vector_l2_ops) where (category_id = 123);

8.4. Partitioning

-- 用分区partitioning对where列的许多不同值进行近似approximate搜索：

create table order_item(
	embedding vector(3),
	category_id int
)partition by list(category_id);

9. Hybrid Search

  • 与Postgres全文搜索合用进行混合搜索；
    https://www.postgresql.org/docs/current/textsearch-intro.html

  • 可用Reciprocal Rank Fusion或cross-encoder来组合结果；
    https://github.com/pgvector/pgvector-python/blob/master/examples/hybrid_search_rrf.py
    https://github.com/pgvector/pgvector-python/blob/master/examples/hybrid_search.py

select id,content from order_item,plainto_tsquery('hello search') query
    where textsearch @@ query order by ts_rank_cd(textsearch,query) desc limit 5;

10.4. Querying

  • ﻿使用explain analyze调试性能

﻿explain analyze select * from order_item
	order by embedding <-> '[3,1,2]' limit 5;

10.4.1. Exact Search

  • 在没索引的情况下加快查询速度，
    请增加参数max_paralle_workers_per_gather

set max_parallel_workers_per_gather = 4;

  • 若向量被标准化为(normalized to)长度1，则使用inner product以获得最佳性能；

select * from order_item order by embedding <#> '[3,1,2]' limit 5;

10.4.2. Approximate Search

  • 要用IVFFlat索引加快查询速度，请增加反向列表的数量inverted list
    number，以召回为代价 at the expense of recall)；

create index on order_item using ivfflat (embedding vector_l2_ops) with (lists = 1000);

10.5. Vacuuming

  • 对HNSW索引进行Vacuuming需一段时间，通过先重新索引来加快速度；

reindex index concurrently index_name;

vacuum table_name;

11. Monitoring

11.1. Monitoring Performance

  • 用pg_stat_statements监控性能，确保其添加到shared_preload_libraries中；

create extension pg_stat_statements;

-- 获取最耗时TIME-CONSUMING的查询：
select query,calls,
	round((total_plan_time + total_exec_time) / calls) as avg_time_ms,
    round((total_plan_time + total_exec_time) / 60000) as total_time_min
    from pg_stat_statements order by total_plan_time + total_exec_time
    desc limit 20;

11.2. Monitoring Recall

  • 通过比较近似approximate搜索和精确exact搜索的结果来监控召回recall；

begin;
-- use exact search
set local enable_indexscan = off;
select ……
commit;

14. Faq

14.1. Common

  • 单表可存储多少向量?：

  • Postgres中非分区表的默认限制为32TB，
    一个分区表(partition)可有数千个这样大小的分区；

  • 是否支持复制?：

  • 支持，Postgres Vector使用预写日志WAL(write-ahead log)，
    它允许复制和时间点恢复replication and point-in-time recovery；

  • 若想索引草果2000个维度的向量该咋办?：

  • 建议降维﻿：use dimensionality reduction；

14.2. Column Mix Dimension

  • 可将不同维度的向量存储在同一列中吗?：

  • 可使用向量作为类型use vector as type (instead of vector(3))；

create table embeddings(
	model_id bigint,
	item_id bigint,
	embedding vector,
	primary key (model_id,item_id)
);

  • 但只能对具有相同维数的行创建索引(使用expression and partial indexing)：

create index on embeddings using
	hnsw ((embedding::vector(3)) vector_l2_ops)
	where (model_id = 123);

-- 查询方式：
select * from embeddings where model_id = 123
	order by embedding::vector(3) <-> '[3,1,2]' limit 5;

14.3. Vector Precision

  • 可更精确precision的存储向量吗?：

  • 可使用double precision[]或numeric[]类型来存储精度更高的向量；

create table order_item(
	id bigserial primary key,
	embedding double precision[]
);

-- use {} instead of [] for Postgres array
insert into order_item (embedding) values('{1,2,3}'),('{4,5,6}');

  • 可选Optionally，添加检查约束，以确保数据可转换为
    向量类型并具有预期的维度expected dimension；

alter table order_item add
	check(vector_dims(embedding::vector) = 3);

  • 使用表达式索引进行索引(以较低的精度)，
    expression indexing to index(at lower precision)；

create index on order_item using
	hnsw ((embedding::vector(3)) vector_l2_ops);

select * from order_item order by embedding::vector(3) <-> '[3,1,2]' limit 5;

14.4. Index Fit Into Memory

﻿* 索引是否需要纳入内存?index need to fit into memory?：

  • 不需要，但与其它索引类型一样，若这样做，可能得到更好的性能，获取索引大小：

select pg_size_pretty(pg_relation_size('index_name'));

15. Troubleshooting

15.1. Index Not Working

  • 为何查询不使用索引：

  • ﻿查询需要有order by和limit，order by必须是距离运算符
    distance operator的结果而非表达式expression；

 
-- index
order by embedding <=> '[3,1,2]' limit 5;

-- no index
order by 1 - (embedding <=> '[3,1,2]') desc limit 5;

-- 鼓励如下方式(planner )查询：
begin;
set local enable_seqscan = off;
select ……
commit;

-- 另若表很小，则表扫描可能会更快

15.2. Parallel Table Scan

  • 为何查询不使用并行表扫描parallel table scan：

  • planner在成本估算中没考虑out-of-line storage，
    这使串行扫描看起来更便宜，可通过以下方式降低查询的并行扫描成本：

  • 或choose to store vector inline；

 
begin;
set local min_parallel_table_scan_size = 1;
set local parallel_setup_cost = 1;
select ……
commit;

-- choose to store vector inline
alter table order_item alter column embedding set storage plain;

15.3. Hnsw Index

  • 为何添加HNSW索引后查询的结果更少there less result?

  • 结果受到动态候选列表大小dynamic candidate list size
    (hnsw.ef_search)的限制，因查询中的dead tuple
    或过滤条件filtering condition，结果可能更少；

  • 建议将hnsw.ef_search设置为至少查询limit的两倍，
    若需500个以上的结果，请使用IVFFlat索引；

15.4. IvfFlat Index

  • 为何添加IVFFlat索引后查询的结果更少there less result?

  • 对列表的数量来说，创建索引的数据可能太少，删除索引，直到表中有更多数据为止；

  • 结果也可能受probe的数量限制(ivfflat.probes)；

 
drop index index_name;

16. Reference

16.1. Vector Type

  • 每个向量占(4 * dimensionNumber +8)字节存储，每个元素都是单精度浮点数，
    如同Postgres中的real type，所有元素都必须是有限的(finite)，
    如没NaN，Infinity，-Infinity，向量最多可有16000个维度；
