7.1. Math Function

  • 要特别注意负数；

import math

number,digit = 10.56789,2

# # 四舍五入和绝对值：10.57 10.56789
print(round(number,digit),abs(number))

# # 向上取整和向下取整：11 10
print(math.ceil(number),math.floor(number))

# 平方根，自然对数的基(natural logarithm base)，自然对数：
# 3.2508291250079573 38866.57866367734 2.357820158384626
print(math.sqrt(number),math.exp(number),math.log(number))

9.4. Modifying String

  • replace，strip，find，index，format；

greeting = "Hello, World!"

# concatenate：Hello, World! Nice meeting you!
print(greeting + ' Nice meeting you!')

# Replace Character，失败，因字符串是immutable
# greeting[0] = 'h'

# Hello, World!How are you?
greeting += 'How are you?'
print("concatenate：",greeting)

# Uppercase： HELLO, WORLD!HOW ARE YOU?
greeting = greeting.upper()
print("Uppercase：",greeting)

# Lowercase： hello, world!how are you?
greeting = greeting.lower()
print("Lowercase：",greeting)

# Splitting： ['hello,', 'world!how', 'are', 'you?']
splitting = greeting.split()
print("Splitting：",splitting)

name,age = 'elf',18

# hello, elf!

print('{0} is {1} years old!'.format(name, age))

print('{1} is {0} years old!'.format(age, name))

10.3. f-strings

name,age = 'elf',18

# hello, elf!
print(f'hello, {name}!')

# elf is 19 years old!
print(f'{name} is {age + 1} years old!')

10.4. Formatting Number

price,pi = 9.99,3.14159
# the price is $9.99 , pi is approximately 3.14
print(f'the price is ${price}',',',f'pi is approximately {pi:.2f}')

variable = 38

# pad with zero 0038
print(f'pad with zero {variable:04d}')

# pad with space       38
print(f'pad with space {variable:8d}')

# pad before with space       38
print(f'pad before with space {variable:>8d}')

# pad after with space 38
print(f'pad after with space {variable:<8d}')

# pad center with space    38
print(f'pad center with space {variable:^8d}')

10.5. Pro Tip

n = 808017424794512875886459904961710757005754368

# number is: 808_017_424_794_512_875_886_459_904_961_710_757_005_754_368
print(f'number is: {n:_d}')

# number is: 808,017,424,794,512,875,886,459,904,961,710,757,005,754,368
print(f'number is: {n:,d}')
million = 1_000_000
billion = 1_000_000_000

# 1000000 1000000000
print(million, billion)

# 1_000_000 1_000_000_000
print(f'{million:_d} {billion:_d}')
